-- Criando o banco de dados
create database db_escola;
-- Especificando o BD
use db_escola;
-- Criando a tabela
create table tb_alunos(
	id bigint(5) auto_increment,
	nome varchar(255) not null,
    idade bigint(3),
    endereco varchar(255),
    serie varchar(255),
    media float(50,2),
    ativo boolean not null,
    primary key(id)
);
